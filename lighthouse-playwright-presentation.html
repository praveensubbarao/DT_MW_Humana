The Problem

Without integration
  • Lighthouse runs manually or in CI as a separate job
  • Performance regressions discovered late
  • Scores live in a separate dashboard — not next to test results
  • No easy way to assert on scores per test
With integration
  • Lighthouse runs as part of every test run
  • Full HTML report attached directly to the test result
  • Scores annotated on the test — visible in the Playwright report
  • Assert thresholds → CI fails on regression

Step 2 — The Core Utility NEW FILE

src/utils/lighthouse/runLighthouse.ts 

export interface LighthouseScores {
  performance: number; accessibility: number;
  bestPractices: number; seo: number;
}

export async function runLighthouse(url: string) {
  const { launch }           = await import('chrome-launcher');
  const { default: lighthouse } = await import('lighthouse');

  // Own Chrome instance — no conflict with Playwright's browser
  const chrome = await launch({
    chromeFlags: ['--headless', '--no-sandbox', '--disable-gpu']
  });
  try {
    const result = await lighthouse(url, {
      port: chrome.port,
      output: ['html', 'json'],          // both formats
      logLevel: 'error',
      onlyCategories: ['performance', 'accessibility', 'best-practices', 'seo'],
    });
    const [htmlReport, jsonReport] = result!.report as [string, string];
    const { lhr } = result!;
    return {
      scores: {
        performance:   Math.round((lhr.categories['performance']?.score   ?? 0) * 100),
        accessibility: Math.round((lhr.categories['accessibility']?.score ?? 0) * 100),
        bestPractices: Math.round((lhr.categories['best-practices']?.score ?? 0) * 100),
        seo:           Math.round((lhr.categories['seo']?.score           ?? 0) * 100),
      },
      htmlReport, jsonReport,
    };
  } finally {
    await chrome.kill();   // always clean up
  }
}

Step 3 — The Playwright Fixture NEW FILE

src/utils/fixtures/lighthouseFixture.ts 

import { test as base, TestInfo } from '@playwright/test';
import { runLighthouse, LighthouseScores } from '@/utils/lighthouse/runLighthouse';

export interface LighthouseActions {
  audit(url: string): Promise<LighthouseScores>;
}

export const lighthouseTest = base.extend<{ lighthouse: LighthouseActions }>({
  lighthouse: async ({}, use, testInfo) => {
    await use({
      audit: async (url: string) => {
        const result = await runLighthouse(url);

        // ① Attach full HTML report → clickable in Playwright HTML report
        await testInfo.attach('Lighthouse Report', {
          body: result.htmlReport,
          contentType: 'text/html',
        });

        // ② Annotate scores → visible in the test detail panel
        const { scores } = result;
        testInfo.annotations.push(
          { type: 'lighthouse:performance',    description: `${scores.performance}`   },
          { type: 'lighthouse:accessibility',  description: `${scores.accessibility}` },
          { type: 'lighthouse:best-practices', description: `${scores.bestPractices}` },
          { type: 'lighthouse:seo',            description: `${scores.seo}`           },
        );
        return scores;
      },
    });
  },
});

Step 5 — Write a Performance Test NEW FILE

src/tests/us/performance/HomepageLighthouse.spec.ts 

import { test, expect } from '@/utils/fixtures/providerBaseTest';

test.describe('Homepage — Lighthouse audit', () => {
  // Steps:
  // 1. Open the homepage via the configured base URL.
  // 2. Run a Lighthouse audit and attach the HTML report.
  // 3. Assert each score meets the defined minimum threshold.
  test('homepage scores meet minimum thresholds',
    async ({ providerActions, lighthouse }) => {   // ← same import, new fixture
      await providerActions.openProviderPortal();

      const scores = await lighthouse.audit(
        providerActions.getBaseURL() + '/'
      );

      expect(scores.performance,    'performance').toBeGreaterThanOrEqual(25);
      expect(scores.accessibility,  'accessibility').toBeGreaterThanOrEqual(70);
      expect(scores.bestPractices,  'best-practices').toBeGreaterThanOrEqual(70);
      expect(scores.seo,            'seo').toBeGreaterThanOrEqual(70);
  });
});

Key Takeaways

🔀 No CDP Conflict

Lighthouse launches its own headless Chrome via chrome-launcher — completely independent from the Playwright browser driving your test.

📎 Report in the Dashboard

testInfo.attach() embeds the full Lighthouse HTML report as a clickable attachment in the Playwright HTML report — no separate dashboard needed.

⚡ Lazy by Design

Lighthouse only runs in tests that call lighthouse.audit(). All other tests are completely unaffected — zero overhead.

🛡️ Quality Gate in CI

Score assertions use standard expect() — a regression causes the test to fail, the CI job fails, and the broken build is blocked from shipping.
